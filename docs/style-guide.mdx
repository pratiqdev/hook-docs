<Tabs>

```jsx
import { useInput } from '@pratiq/hooks'

const searchInput = useInput({placeholder: 'Search...', value: 'cute cats'})

```

</TabItem>

<TabItem value='advanced' label='Advanced'>

```jsx
import { useInput } from '@pratiq/hooks'

const searchValidator = (val) => {
  return val.length >= 4
}

const searchInput = useInput({
  name: 'search-input', 
  placeholder: 'Search...', 
  value: '',
  validator: searchValidator,
  validateOnChange: true,
  validOnDefault: false,
})

<input {...searchInput.bind}/>

<button onClick={ searchInput.reset }>
  Reset Array to Initial State
</button>

```

</TabItem>

## Configuration

Values used for initial state and internal settings are provided as a config
object 


### Interface
```ts
interface I_useInputConfig {
    type?: string;
    name?: string;
    value?: string;
    placeholder?: string;
    readOnly?: boolean;
    disabled?: boolean;
    validOnDefault?: boolean;
    validateOnBlur?: boolean;
    validateOnChange?: boolean;
    validator?: (a: string) => boolean;
    onHover?: (e: event) => void;
    onFocus?: (e: event) => void;
    onBlur?: (e: event) => void;
}
```


### Config Example

```ts
{
  type: 'text',
  name: 'input-name',
  value: 'Initial Value',
  placeholder: 'Enter value here...',
  readOnly: false;
  disabled: false;
  validOnDefault: false;
  validateOnBlur: true;
  validateOnChange: true;
  validator: (val) => return val.length >= 4 ? true : false,
  onHover: () => triggerAnEvent()
  onFocus: () => triggerAnEvent()
  onBlur: () => triggerAnEvent()
}
```
